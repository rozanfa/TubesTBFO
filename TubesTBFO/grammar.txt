/* MAIN PROGRAM */
START -> GRAMMAR
GRAMMAR -> GRAMMAR ENDL GRAMMAR | GRAMMAR ENDL | ENDL GRAMMAR | FOR_LOOP_GRAMMAR | WHILE_LOOP_GRAMMAR | IMPORT_GRAMMAR | CONDITIONAL_GRAMMAR | INT | ITERATABLE


===================================================================================================================================================
/* TRUE FALSE */
TRUE -> n
FALSE -> o

/* NOT IS IN */
NOT -> p
IS -> q
IN -> r

/ *OR AND */
OR -> s
AND -> t

/* CLASS */
CLASS -> u
CLASS_GRAMMAR -> CLASS_HEADER CLASS_BODY
CLASS_HEADER -> CLASS ANY TITIK_DUA ENDL | CLASS ANY KURUNG_BUKA CLASS_PARAM KURUNG_TUTUP TITIK_DUA ENDL
CLASS_PARAM -> CLASS_PARAM KOMA CLASS_PARAM | ANY
CLASS_BODY -> CLASS_BODY ENDL CLASS_BODY | CLASS_BODY ENDL | IF_GRAMMAR  | CLASS_GRAMMAR | DEF_FUNC_GRAMMAR |

METHOD_GRAMMAR -> METHOD_INIT TITIK METHOD_GRAMMAR | METHOD_BACK_FUNC | METHOD_BACK_NAME
METHOD_GRAMMAR_NAME -> METHOD_INIT TITIK METHOD_GRAMMAR_NAME | METHOD_BACK_NAME
METHOD_GRAMMAR_FUNC -> METHOD_INIT TITIK METHOD_GRAMMAR_FUNC | METHOD_BACK_FUNC
METHOD_INIT -> KURUNG_BUKA METHOD_INIT KURUNG_TUTUP | METHOD_BACK_FUNC | METHOD_BACK_NAME
METHOD_BACK_FUNC -> FUNCTION
METHOD_BACK_NAME -> ANY

/* FUNCTION */
DEF -> v
DEF_FUNC_GRAMMAR -> DEF_FUNC_HEADER DEF_FUNC_BODY
DEF_FUNC_HEADER -> DEF ANY KURUNG_BUKA DEF_FUNC_PARAM KURUNG_TUTUP TITIK_DUA ENDL | DEF ANY KURUNG_BUKA KURUNG_TUTUP TITIK_DUA ENDL
DEF_FUNC_BODY -> DEF_FUNC_BODY ENDL DEF_FUNC_BODY | GRAMMAR_FUNC
DEF_FUNC_RETURN ->  RETURN DEF_FUNC_RETURN_VAL | RETURN
DEF_FUNC_RETURN_VAL -> FUNCTION | VARIABEL | STRING
DEF_FUNC_PARAM -> DEF_FUNC_PARAM KOMA DEF_FUNC_PARAM | ANY | ANY TITIK_DUA ANY
RETURN -> w

// pemanggilan fungsi Function
FUNCTION -> FUNCTION_BASE
FUNCTION_BASE -> VARIABEL KURUNG_BUKA FUNCTION_PARAM KURUNG_TUTUP | VARIABEL KURUNG_BUKA KURUNG_TUTUP
FUNCTION_PARAM -> FUNCTION_PARAM KOMA FUNCTION_PARAM | VARIABEL | STRING | FUNCTION


/* STRING */
// String
STRING -> x | STRING KALI POSITIVE_NUMBER | STRING TAMBAH STRING

// ANY
ANY -> z

===================================================================================================================================================
/* IMPORT */

IMPORT_GRAMMAR -> IMPORT_GRAMMAR ENDL IMPORT_GRAMMAR | IMPORT LIB ENDL | FROM LIB IMPORT LIB_GRAMMAR ENDL | IMPORT LIB AS VAR ENDL | FROM LIB IMPORT GRAMMAR AS VAR ENDL | IMPORT MULTIPLE_IMPORT
MULTIPLE_IMPORT -> MULTIPLE_IMPORT KOMA MULTIPLE_IMPORT | LIB AS VAR 
LIB_GRAMMAR -> LIB_GRAMMAR KOMA LIB_GRAMMAR | GRAMMAR
IMPORT -> import
LIB -> ANY
GRAMMAR -> ANY


/* FOR LOOP */

FOR_LOOP_GRAMMAR -> FOR_LOOP_HEADER CODE_GRAMMAR 
FOR_LOOP_HEADER -> FOR VAR IN ITERATABLE TITIK_DUA ENDL

FOR -> for
IN -> in
ITERATABLE -> RANGE_FUNC | TUPLE | LIST | STRING
RANGE_FUNC -> RANGE KURUNG_BUKA INT KURUNG_TUTUP

/* WHILE LOOP */

WHILE_LOOP_GRAMMAR -> WHILE_LOOP_HEADER CODE_GRAMMAR
WHILE_LOOP_HEADER -> WHILE BOOL_EXPRESSION TITIK_DUA ENDL


INT -> VAR | NUMBER


/* CONDITIONAL */

CONDITIONAL_GRAMMAR -> IF_HEADER CODE_GRAMMAR | IF_HEADER CODE_GRAMMAR ELSE_CONDITION
IF_HEADER -> IF BOOL_EXPRESSION TITIK_DUA ENDL
IF -> if

ELSE_CONDITION -> ELSE_GRAMMAR | ELIF_GRAMMAR ELSE_GRAMMAR

ELIF_GRAMMAR -> ELIF_GRAMMAR ELIF_GRAMMAR | ELIF_HEADER CODE_GRAMMAR
ELIF_HEADER -> ELIF | BOOL_EXPRESSION TITIK_DUA ENDL
ELIF -> elif

ELSE_GRAMMAR -> ELSE_HEADER CODE_GRAMMAR
ELSE_HEADER -> ELSE | BOOL_EXPRESSION TITIK_DUA ENDL
ELSE -> else

/* RAISE */
RAISE_GRAMMAR -> RAISE EROR_TYPE RAISE_CONTENT
EROR_TYPE -> ANY
RAISE_CONTENT ->  KURUNG_BUKA RAISE_CONTETNT KURUNG_TUTUP | VARIABLE | CONSTANT | FUNCTION | LIST | STRING | DICT | NONE | COND_OP | ARITMETHIC_OP | METHOD_GRAMMAR
RAISE -> n

/* OPEN_FILE */
OPEN_FILE_GRAMMAR -> WITH_HEADER CONTENT
WITH_HEADER -> WITH WITH_MODE AS VARIABEL TITIK_DUA ENDL
WITH_MODE -> FUNCTION | METHOD_GRAMMAR_FUNC
WITH -> o

/* VARIABEL */
VARIABEL -> KURUNG_BUKA VARIABEL KURUNG_TUTUP | VAR_SINGLE
VAR_SINGLE -> NAME | NAME IDX_VARIABLE | METHOD_GRAMMAR_NAME | METHOD_GRAMMAR_NAME IDX_VARIABLE
IDX_VARIABLE -> IDX_VARIABLE IDX_VARIABLE | [ IDX ] | [ TITIK_DUA ] | [ IDX TITIK_DUA ] | [ TITIK_DUA IDX ] | [ IDX TITIK_DUA IDX ] | [ TITIK_DUA TITIK_DUA ] | [ TITIK_DUA TITIK_DUA IDX ] | [ TITIK_DUA IDX TITIK_DUA ] | [ TITIK_DUA IDX TITIK_DUA IDX ] | [ IDX TITIK_DUA TITIK_DUA ] | [ IDX TITIK_DUA TITIK_DUA IDX ] | [ IDX TITIK_DUA IDX TITIK_DUA ] | [ IDX TITIK_DUA IDX TITIK_DUA IDX ] 
IDX -> VARIABEL | NUMBER | CONSTANT

/* NUMBER */
NUMBER -> TAMBAH NUMBER | KURANG NUMBER | NUM_SINGLE
POSITIVE_NUMBER -> TAMBAH POSITIVE_NUMBER | NUM_SINGLE
NEGATIVE_NUMBER -> KURANG NEGATIVE_NUMBER | NUM_SINGLE
NUM_SINGLE -> p
CONSTANT -> KURUNG_BUKA CONSTANT KURUNG_TUTUP | CONS_SINGLE
CONS_SINGLE -> TRUE | FALSE | NUMBER

/* LIST */
LIST -> KURUNG_SIKU_BUKA KURUNG_SIKU_TUTUP | KURUNG_SIKU_BUKA LIST_CONTENTS KURUNG_SIKU_TUTUP
LIST_CONTENTS -> LIST_TYPE | LIST_TYPE FOR VARIABEL IN FUNCTION | LIST_CONTENTS KOMA LIST_TYPE
LIST_TYPE -> STRING | VARIABEL | CONSTANT | DICT | NONE | FUNCTION | METHOD_GRAMMAR | LIST 

/* DICT */
DICT -> KURUNG_KURAWAL_BUKA KURUNG_KURAWAL_TUTUP | KURUNG_KURAWAL_BUKA DICT_CONTENTS KURUNG_KURAWAL_TUTUP
DICT_CONTENTS -> DICT_CONTENT | DICT_CONTENT KOMA DICT_CONTENT
DICT_CONTENT -> DICT_TYPE TITIK_DUA DICT_TYPE
DICT_TYPE -> STRING | VARIABEL | CONSTANT | DICT | NONE | FUNCTION | METHOD_GRAMMAR | LIST
